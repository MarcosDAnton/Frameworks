Soma por Telefone com o chefe maluco da cabeça.

Algoritmo Soma-Por-Telefone {

  inteiro: NUM [50], SOMA, i, Valor, P;

  SOMA <- 0;
  P <- 0;

  para i de 0 a 999 {
    leia(Valor);

     se (Valor > 0) {

       NUM [P] <- Valor; 
       SOMA <- SOMA + Valor;
       P <- P + 1;
       }
         senão se (valor == 0 e P > 0) {
             P <- P - 1; 
             SOMA <- SOMA - NUM [P];
         }
     }
      escreva ("Somatório: ", SOMA);
